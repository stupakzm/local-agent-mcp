    ```typescript
    export interface LoopStep {
      toolName: string;
      args: Record<string, unknown>;
      result: ToolResult;
    }

    export interface AgentResult {
      steps: LoopStep[];
      finalMessage: string;
      iterationCount: number;
      stoppedByLimit: boolean;
    }
    ```

    **Main function:**
    ```typescript
    export async function runAgentLoop(options: {
      prompt: string;
      model: string;
      host: string;
      workingDir: string;
      maxIterations: number;
      shellMode: ShellMode;
      allowedCommands: readonly string[];
      timeoutMs: number;
    }): Promise<AgentResult>
    ```

    **Loop algorithm:**
    1. Initialize `messages: OllamaMessage[]` with a system message:
       ```
       "You are a helpful coding assistant. You have access to tools for reading files, writing files, listing directories, and executing shell commands. Use these tools to complete the user's task. When you are done, respond with a final text message summarizing what you did."
       ```
    2. Append user message: `{ role: "user", content: prompt }`
    3. Initialize `steps: LoopStep[] = []`, `iteration = 0`
    4. **Loop:** while `iteration < maxIterations`:
       a. `iteration++`
       b. Call `chatWithOllama(host, { model, messages, tools: TOOL_DEFINITIONS, stream: false })`
       c. Get `response.message`
       d. **CRITICAL (LOOP-04):** Append the assistant message to `messages` BEFORE processing tool results:
          `messages.push(response.message)`
       e. Check `response.message.tool_calls`:
          - If undefined, empty, or length 0: **done** -- break loop, set `finalMessage = response.message.content`
          - If present: iterate over each tool call:
            1. Extract `tc.function.name` and `tc.function.arguments` (already parsed object, do NOT JSON.parse)
            2. Call `executeTool(name, arguments, workingDir, shellMode, allowedCommands, timeoutMs)`
            3. Record step: `{ toolName: name, args: arguments, result }`
            4. **(LOOP-03):** Append tool result to messages -- even on error:
               `messages.push({ role: "tool", content: result.output })`
    5. After loop exits:
       - If `iteration >= maxIterations` and last message had tool_calls: set `stoppedByLimit = true`, `finalMessage = response.message.content || ""` -- the model's last partial response
       - Otherwise: `stoppedByLimit = false`
    6. Return `{ steps, finalMessage, iterationCount: iteration, stoppedByLimit }`

    **Error handling:**
    - If `chatWithOllama` throws (Ollama not running, network error): let it propagate -- the MCP handler in index.ts will catch it and return isError
    - Tool execution errors are already handled by executeTool returning `{ success: false }` -- never thrown
    - Log each iteration to stderr: `console.error(\`[agent] iteration ${iteration}: ${toolCalls.length} tool call(s)\`)`

    Import from `./ollama.js`, `./tools.js`, `./security.js`. All ESM .js extensions.

    ```typescript
    const logLines: string[] = [];

    for (const step of result.steps) {
      const argsStr = Object.entries(step.args)
        .map(([k, v]) => `${k}=${typeof v === "string" && v.length > 80 ? v.slice(0, 80) + "..." : JSON.stringify(v)}`)
        .join(", ");

      if (step.result.success) {
        // Summarize output: line count for files, exit code for bash
        const summary = step.result.output.length > 200
          ? `${step.result.output.split("\n").length} lines`
          : step.result.output.trim();
        logLines.push(`${step.toolName}(${argsStr}) → ${summary}`);
      } else {
        logLines.push(`${step.toolName}(${argsStr}) → blocked: ${step.result.output}`);
      }
    }

    if (result.stoppedByLimit) {
      logLines.push(`[stopped: max iterations reached (${MAX_ITERATIONS})]`);
    }

    const executionLog = logLines.length > 0
      ? logLines.join("\n") + "\n\n"
      : "";

    const responseText = executionLog + result.finalMessage;

    return {
      content: [{ type: "text" as const, text: responseText }],
    };
    ```
